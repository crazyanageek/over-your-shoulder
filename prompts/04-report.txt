=== STEP 2 — Update manifest.json ===

Add to the manifest:
- web_accessible_resources: expose /report/* to extension pages (so the report can be opened as a chrome-extension:// URL)
- content_security_policy.extension_pages: allow fonts.googleapis.com and fonts.gstatic.com (for Google Fonts)
  Specifically: "script-src 'self'; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com; connect-src 'self';"

The 'unsafe-inline' for style-src is necessary because D3 sets inline styles on SVG elements at runtime. Script-src stays 'self' only — no remote scripts.

=== STEP 6 — /report/report.js — the rendering logic ===

The report runs in the extension context (chrome-extension:// URL) so it has full access to chrome.storage.local. No message passing needed.

The logic flow:

async function initReport() {
  // 1. Read all storage data
  const all = await chrome.storage.local.get(null);
  const counters = all.oys_counters || defaultCounters();
  const daysActive = countDaysActive(all);

  // 2. Compute the score (import or copy from /lib/score.js)
  const { score, breakdown } = computeScore(counters, daysActive);

  // 3. Load punchlines config
  const punchlinesURL = chrome.runtime.getURL('report/data/punchlines.json');
  const punchlines = await fetch(punchlinesURL).then(r => r.json());

  // 4. Pick punchlines (random at each open)
  const lang = 'en';
  const masthead = pickRandom(punchlines.masthead[lang]);
  const heroSub = pickRandom(punchlines.heroSub[lang]);
  const verdict = pickVerdict(punchlines.verdict[lang], score);

  // 5. Pick map headline based on geographic distribution
  const totalRequests = counters.total || 0;
  const usShare = (counters.byCountry?.US || 0) / Math.max(1, totalRequests);
  const mapHeadlineTemplate = usShare >= 0.8
    ? pickRandom(punchlines.mapHeadline[lang].usDominant)
    : pickRandom(punchlines.mapHeadline[lang].neutral);
  const countryCount = Object.keys(counters.byCountry || {}).length;
  const usTotal = counters.byCountry?.US || 0;
  const mapHeadline = mapHeadlineTemplate
    .replace('{N}', usDominant ? usTotal.toLocaleString() : totalRequests.toLocaleString())
    .replace('{C}', countryCount);

  // 6. Render everything into the DOM
  renderMasthead(masthead, computeDateline(daysActive));
  renderHero(totalRequests);   // puts the big number
  renderHeroSub(heroSub);
  renderScore(score, breakdown, verdict);
  renderMap(counters);         // uses D3 + world-110m.json
  renderStats(counters, totalRequests);
  renderFooter();

  // 7. Fit hero number dynamically (shrink if it overflows)
  fitHeroNumber();
}

Key helpers:

function pickRandom(arr) {
  return arr[Math.floor(Math.random() * arr.length)];
}

function pickVerdict(verdictObj, score) {
  let band;
  if (score <= 20) band = 'low';
  else if (score <= 40) band = 'mild';
  else if (score <= 60) band = 'moderate';
  else if (score <= 80) band = 'high';
  else band = 'alarming';
  return pickRandom(verdictObj[band].lines);
}

function computeDateline(daysActive) {
  const d = new Date();
  const weekday = d.toLocaleDateString('en-US', { weekday: 'short' });
  const day = d.getDate();
  const month = d.toLocaleDateString('en-US', { month: 'short' });
  const dateStr = `${weekday} ${day} ${month}`;
  const daysStr = daysActive === 1 ? '1 day of data' : `${daysActive} days of data`;
  return { dateStr, daysStr };
}

Edge cases:
- If totalRequests is 0: show hero number as "—" and replace sub-text with "No data yet. Come back tomorrow."
- If daysActive === 0: still render, but verdict reads "No signal yet".
- If punchlines.json fails to load: fall back to hardcoded English defaults (one per zone) and log a warning to console.
